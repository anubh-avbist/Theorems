\textbf{1. } Suppose $f:[a,b]\rightarrow \mathbb{R}$ is a bounded function such that $L(f,P,[a,b])=U(f,P,[a,b])$ for some partition $P$ of $[a,b]$. Prove that $f$ is the constant function on $[a,b]$.
\newline
\textit{Proof.} Let $a=x_0,\ldots,x_n=b$ denote the partition $P$ of $[a,b]$. Since the lower and upper Riemann sums are equal, we have
\begin{align*}
    \sum_{j=1}^n(x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\sup}f=\sum_{j=1}^n(x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\inf}f.\\
    \implies
    \sum_{j=1}^n(x_j-x_{j-1})\Big(\underset{[x_{j-1},x_j]}{\sup}f-\underset{[x_{j-1},x_j]}{\inf}f\Big)=0\\
\end{align*}
The length of each interval is strictly positive, and $\underset{[x_{j-1},x_j]}{\sup}f\geq \underset{[x_{j-1},x_j]}{\inf}f$ for each $j$, so the above sum consists of only non-negative terms. Thus, we have $(x_j-x_{j-1})(\underset{[x_{j-1},x_j]}{\sup}f-\underset{[x_{j-1},x_j]}{\inf}f)=0$ for each interval, and $(\underset{[x_{j-1},x_j]}{\sup}f=\underset{[x_{j-1},x_j]}{\inf}f)$. Thus, $f$ is constant on every interval $[x_{j-1},x_j]$.
\\
\newline
\newline
$f(x)=f(a)$ for every $x\in[x_1,x_2]$. Suppose $f$ is $f(a)$ on an interval $[x_{j-1},x_j]$. Since $f$ is also constant on the interval $[x_j,x_{j+1}]$, $f(x)=f(x_j)=f(a)$ for all $x\in [x_j,x_{j+1}]$. By mathematical induction, $f(x)=f(a)$ on every interval $[x_j,x_{j+1}]$. Therefore, $f$ is constant on the entire domain.\qed

\textbf{2. } Suppose that $a\leq s < t \leq b$. Define $f:[a,b]\rightarrow \mathbb{R}$ by 
\begin{equation*}
    f(x)=
    \begin{cases*}
        1 &\text{if }$s<x<t$,\\
        0 &\text{otherwise.}
    \end{cases*}
\end{equation*}
Prove that $f$ is Riemann integrable on $[a,b]$ and that $\int_{a}^{b}f=t-s$. 
\newline
\textit{Proof.} Consider the partition $P$ of $[a,b]$ into $n\in\mathbb{N}$ intervals of equal length given by $x_i=a+i\frac{b-a}n$. Notice that 
\begin{equation*}
    L(f,P,[a,b])=\sum_{i=1}^{n}d\underset{[x_{i-1},x_i]}{\inf}f\geq(\floor{\frac{n(t-s)}{b-a}}-2)\frac{b-a}{n},
\end{equation*}
and similarly
\begin{equation*}
    U(f,P,[a,b])=\sum_{i=1}^{n}d\underset{[x_{i-1},x_i]}{\sup}f\leq\floor{\frac{n(t-s)}{b-a}}\frac{b-a}{n}.
\end{equation*}
Then,
\begin{equation*}
    L(f,[a,b])
    \geq \underset{n\in \mathbb{N}}{\sup}(\floor{\frac{n(t-s)}{b-a}}-2)\frac{b-a}{n}
    = t-s
    = \underset{n\in \mathbb{N}}{\inf}(\floor{\frac{n(t-s)}{b-a}})\frac{b-a}{n}
    \geq U(f,[a,b]).
\end{equation*}
But, since $L(f,[a,b])\leq U(f,[a,b])$ we have $L(f,[a,b])= U(f,[a,b])=t-s$, and hence $\int_{a}^{b}f=t-s$.\qed

\textbf{3. }Suppose $f:[a,b]\rightarrow \mathbb{R}$ is a bounded function. Prove that $f$ is Riemann integrable if and only if for each $\epsilon >0$ there exists a partition $P$ of $[a,b]$ such that 
\begin{equation*}
    U(f,P,[a,b])-L(f,P,[a,b])<\epsilon.
\end{equation*}
\textit{Proof.} \\
($\implies$) Suppose $f$ is Riemann integrable, and let $I:=\int_{a}^{b}f$. Let $\epsilon >0$. Since $I=\underset{P}\sup L(f,P,[a,b])=\underset{P}\inf U(f,P,[a,b])$, there is a partition $P_1$ so that $L(f,P_1,[a,b])>I-\frac\epsilon 2$ and a partition $P_2$ so that $U(f,P_2,[a,b])<I+\frac{\epsilon}2$. Let $P$ be the partition obtained by taking the union of $P_1$ and $P_2$. Since $P$ is a finer partition than both $P_1$ and $P_2$, we have
\begin{equation*}
    L(f,P,[a,b])\geq L(f,P_1,[a,b])>I-\epsilon/2
\end{equation*}
and
\begin{equation*}
    U(f,P,[a,b])\leq U(f,P_2,[a,b])<I+\epsilon/2.
\end{equation*}
\noindent
Hence, $|L(f,P[a,b])-I|<\epsilon/2$ and $|U(f,P[a,b])-I|<\epsilon/2$. By the triangle inequality, we have 
\begin{align*}
    U(f,P,[a,b])-L(f,P,[a,b])&=\leq|U(f,P,[a,b])-L(f,P,[a,b])|\\
    &=|U(f,P,[a,b])-I+I-L(f,P,[a,b])|\\
    &\leq|U(f,P,[a,b])-I|+|I-L(f,P,[a,b])|\\
    &<2(\epsilon/2)=\epsilon,
\end{align*}
as desired.
\newline
\newline
($\impliedby$) Suppose that $f$ is not Riemann integrable. Let $L:=L(f,[a,b])$ and $U:=U(f,[a,b])$ be the lower and upper Riemann integrals, where $L\neq U$. Since $L(f,P,[a,b])\leq U(f,P,[a,b])$ for any partition, $L\leq U$. Let $\epsilon:=\frac{U-L}2$. Then, for any partition $P$ of $[a,b]$, we have 
\begin{align*}
    U(f,P,[a,b])-L(f,P,[a,b])&=\geq \underset{P}{\inf}U(f,P,[a,b])-\underset{P}{\sup}L(f,P,[a,b])\\
    &= U-L\\
    &>\epsilon.
\end{align*}
\qed

\textbf{4. }Suppose $f,g:[a,b]\rightarrow \mathbb{R}$ are Riemann integrable. Prove that $f+g$ is Riemann integrable and that $\int_{a}^{b}(f+g)=\int_{a}^{b}f+\int_{a}^{b}g$. 
\newline
\textit{Proof.} Recall that the infimum and supremum distribute over the minkowski sums of sets. Then, we have
\begin{align*}
    L(f+g,[a,b])&=\underset{P}{\sup} L(f+g,P,[a,b])\\
    &=\underset{P}{\sup} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\inf} f+g\\
    &=\underset{P}{\sup} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\inf} f +\underset{P}{\sup} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\inf} f\\
    &= \int_{a}^{b}f+\int_{a}^{b}g\\
    &= \underset{P}{\inf} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\sup} f +\underset{P}{\inf} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\sup} f\\
    &=\underset{P}{\inf} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\sup} f+g\\
    &=\underset{P}{\inf} U(f+g,P,[a,b])\\
    &=U(f+g,[a,b])\\
\end{align*}
Hence, $L(f+g,[a,b])=U(f+g,[a,b])=\int_{a}^{b}f+\int_{a}^{b}g$. \qed

\textbf{5. }Suppose $f:[a,b]\rightarrow \mathbb{R}$ is Riemann integrable. Prove that $-f$ is Riemann integrable and that $\int_{a}^{b}(f)=-\int_{a}^{b}f$. 
\newline
\textit{Proof.} 
\begin{align*}
    L(-f,[a,b])&=\underset{P}{\sup} \sum_{j=1}^{n} (x_j-x_{j-1})\underset{[x_{j-1},x_j]}{\inf} (-f)\\
    &=-\underset{P}{\inf} -\sum_{j=1}^{n} (x_j-x_{j-1})(-\underset{[x_{j-1},x_j]}{\sup} (f))\\
    &=-\underset{P}{\inf} \sum_{j=1}^{n} (x_j-x_{j-1})(\underset{[x_{j-1},x_j]}{\sup} (f))\\
    &=-\int_{a}^{b}f\\
    &=-\underset{P}{\sup} \sum_{j=1}^{n} (x_j-x_{j-1})(\underset{[x_{j-1},x_j]}{\inf} (f))\\
    &=\underset{P}{\inf} -\sum_{j=1}^{n} (x_j-x_{j-1})(-\underset{[x_{j-1},x_j]}{\sup} (-f))=U(-f,[a,b])
\end{align*}
\qed

\textbf{6.} Suppose $f:[a,b]\rightarrow \mathbb{R}$ is Riemann integrable. Suppose $g:[a,b]\rightarrow \mathbb{R}$ is a function such that $g(x)=f(x)$ for all except finitely many $x\in[a,b]$. Prove that $g$ is Riemann integrable on $[a,b]$ and 
\begin{equation*}
    \int_{a}^{b}g=\int_{a}^{b}g.
\end{equation*}
\textit{Proof.} Let $h:=g(x)-f(x)$ be a function defined on $[a,b]$. Since $g=f+h$, and $f$ is Riemann integrable, it suffices to show that $h$ is Riemann integrable. Let $x_1,...,x_n$ be the collection of all finitely many points so that $g(x_i)\neq f(x_i)$ for all $i$, given in increasing order. Moreover, let $C:=\underset{i=1,...,n}{\text{sup}}h(x_i)-\underset{i=1,...,n}{\text{inf}}h(x_i)$.
\newline
\newline
Let $\epsilon>0$, and $N:=\ceil{\frac{b-a}\epsilon}$. Then, let $P_1:=\{a=y_0,y_1,...,y_N=b\}$ where $y_i=\text{min}\{a+i\epsilon,b\}$, and let $P_2:=\{a,x_1,...,x_n,b\}$. Consider the finer partition $P=P_1\cup P_2$, whose elements shall be indexed $a=z_0,z_1,z_2,...,z_M=b$. Then, we have
\begin{align*}
    U(h,P,[a,b])-L(h,P,[a,b])&= \sum_{i=1}^{M}(z_{i}-z_{i-1})\underset{[z_{i-1},z_i]}{\sup} h- \sum_{i=1}^{M}(z_{i}-z_{i-1})\underset{[z_{i-1},z_i]}{\inf} h\\
    &\leq \sum_{i=1}^{M}(z_{i}-z_{i-1})|\underset{[z_{i-1},z_i]}{\sup} h-\underset{[z_{i-1},z_i]}{inf} h|\\
    &\leq \epsilon\sum_{i=1}^{M}|\underset{[z_{i-1},z_i]}{\sup} h-\underset{[z_{i-1},z_i]}{inf} h|\\
\end{align*}
Notice that $\sup h$ is identically 0 on every interval $[z_{i-1},z_i]$ that does not contain some $x_j$. Let $m_1,...,m_n$ be the indices so that $z_{m_j}=x_j$ for every $1\leq j \leq n$. Then,
\begin{align*}
    \epsilon\sum_{i=1}^{M}|\underset{[z_{i-1},z_i]}{\sup} h-\underset{[z_{i-1},z_i]}{inf} h|&=\epsilon\sum_{i=1}^{n}|\underset{[z_{m_i-1},z_{m_i}]}{\sup} h-\underset{[z_{m_i-1},z_{m_i}]}{inf} h|\\
    &\leq \epsilon \cdot(nC)
\end{align*}
Since $nC$ is constant and $\epsilon$ was arbitrary, we have found a partition so that $U(h,P,[a,b])-L(h,P[a,b])\leq \epsilon$ for each $\epsilon$. Hence, $h$ is Riemann integrable, and consequently, so is $g$. \qed

\textbf{7. }Suppose $f:[a,b]\rightarrow \mathbb{R}$ is a bounded function. For $n\in \mathbb{N}$, let $P_n$ denote the partition that divides $[a,b]$ into $2^n$ intervals of equal size. Prove that 
\begin{equation*}
    L(f,[a,b])=\underset{n\rightarrow\infty}{\lim} L(f,P_n,[a,b]) \text{ and }U(f,[a,b])=\underset{n\rightarrow\infty}{\lim} U(f,P_n,[a,b])
\end{equation*}
\textit{Proof. }We prove the statement for the lower integral. Let $\epsilon >0$. Since $L(f,[a,b])=\underset{P}\sup L(f,P,[a,b])$, there is some partition $P$ so that $|L(f,[a,b])-L(f,P,[a,b])|<\epsilon$. Let $a=x_0,x_1,...,x_m=b$ denote the partition of $P$. Let $\delta = \min\{(x_i-x_{i-1}:1\leq i \leq m)\}\cup\{\epsilon\}$. Let $N\in\mathbb{N}$ be so that $\frac{b-a}{N}<\delta$. Then, where $n>N$, consider the partition $Q=P\cup P_n$. 
\newline
\newline
Since $Q$ is a refinement of $P$, we have $L(f,P,[a,b])\leq L(f,Q,[a,b])\leq L(f,[a,b])$. Moreover, since $L(f,[a,b])-L(f,P,[a,b])<\epsilon$, we have $L(f,Q,[a,b])-L(f,P,[a,b])<\epsilon$. 
\newline
\newline
Let $M>0$ be so that $|f(x)|<M$ for all $x\in[a,b]$ (since $f$ is bounded). $P_n$ consists of intervals of size at most $\delta$. $Q$ is a refinement of $P_n$ obtained by adjoining the $m$ points of $P$, each of which splits one of the $\delta$-sized intervals. Each of these insertions contributes a difference of at most $\delta M$ from the lower sum of $L(f,P_n,[a,b])$. Thus, altogether, we have $|L(f,Q,[a,b])-L(f,P_n,[a,b])|\leq m\delta M$. Thus,
\begin{align*}
    &|L(f,P_n,[a,b])-L(f,[a,b])|\\
    &=|L(f,P_n,[a,b])-L(f,Q,[a,b])+L(f,Q[a,b])-L(f,P,[a,b])+L(f,P,[a,b])-L(f,[a,b])|\\
    &\leq|L(f,P_n,[a,b])-L(f,Q,[a,b])|+|L(f,Q[a,b])-L(f,P,[a,b])|+|L(f,P,[a,b])-L(f,[a,b])|\\
    &\leq m\delta M + \epsilon + \epsilon\\
    &\leq\epsilon(mM+2).
\end{align*}
Since $mM+2$ is constant and $\epsilon$ was arbitrary, we have found $N$ so that when $n>N$, we have $L(f,P_n,[a,b])$ is arbitrarily close to $L(f,[a,b])$. Hence, $ L(f,[a,b])=\underset{n\rightarrow\infty}{\lim} L(f,P_n,[a,b]) $. \qed
